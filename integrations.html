quickstart.py
from veritas import Veritas

client = Veritas(api_key="vk_live_8f…2a91")

# create a market
market = client.markets.create(
    question="Did BTC close above $100k on 2026-05-20?",
    primary_authority="coinmarketcap.com",
    threshold=0.90,
    chains=["arbitrum", "base"],
)

# read a resolution + its full reasoning
res = client.resolutions.get("R-4471")
print(res.verdict, res.confidence)   # "YES" 0.942
print(res.ipfs_report)               # full council record
